Promises and Fetch

In this lesson, I learnt about promises to perform API requests to a server. Promises can also be used to run functions as well where we would like to perform a certain action when a successful or failed result is returned from the promise.

Code:

  1. let app = {};
  2.  
  3. /**
  4.  * To define a promise we use the Promise() class which always has two arguments:
  5.  * 1. resolve - this means the promise finished running and return a successful response
  6.  * 2. reject- the promise finished running and something went wrong
  7.  * 
  8. */
  9. let example_promise = new Promise((resolve, reject) => {
  10.     setTimeout(() => {
  11.         resolve();
  12.     }, 1000);     
  13. });
  14.  
  15. /**
  16.  * Here, we get to see that the then() callback is run only when the promise is resolved
  17.  * and catch is run when the promise is rejected
  18. */
  19. example_promise
  20.     .then(() => {
  21.         console.log('Promise result: Finally finished');
  22.     })
  23.     .then(() => {
  24.         console.log('Promise result: I will also run');
  25.     })
  26.     .catch(() => {
  27.         console.log('Promise result: I only run when rejected');
  28.     });
  29.  
  30. /**
  31.  * Let's run it
  32. */
  33. example_promise;
  34.  
  35. const example_url = 'https://jsonplaceholder.typicode.com/todos/1';
  36. /**
  37.  * Below we are using the Fetch API provided in JavaScript to perform a
  38.  * HTTP request. The fetch method is a promise based HTTP client that
  39.  * we can use to perform basic HTTP requests like below. However fetch() has
  40.  * limitations such as catching HTTP errors from the server.
  41. */
  42. const example_fetch = fetch(example_url)
  43.     .then((response) => response.json())
  44.     .then(data => {
  45.         console.log(`Example fetch result: ${JSON.stringify(data)}`);
  46.     });
  47.  
  48. example_fetch;
  49.  
  50. /**
  51.  * We can also use another promise based HTTP client called axios which has
  52.  * far better handling of requests.
  53. */
  54. const axios_url = 'https://jsonplaceholder.typicode.com/posts/1/comments';
  55. axios.get(axios_url)
  56.     .then(response => {
  57.         const comments = response.data;
  58.         console.log(`Axios comments result: ${JSON.stringify(comments)}`);
  59.     })
  60.     .catch(error => {
  61.         console.log(error.response);
  62.     });
  63.  
  64. /**
  65.  * We can further use a promise to perform all get requests where we have axios
  66.  * receiving urls dynamically.
  67. */
  68. const perform_get_request = (request_url) => {
  69.     return new Promise((resolve, reject) => {
  70.         axios.get(request_url)
  71.             .then(response => {
  72.                 resolve(response.data);
  73.             })
  74.             .catch(error => {
  75.                 reject(error.response);
  76.             });
  77.     });
  78. };
  79.  
  80. let names = [];
  81. let user_count = 0;
  82. const users_url = 'https://jsonplaceholder.typicode.com/users';
  83.  
  84. /**
  85.  * In this example we reuse the above promise to perform our request
  86.  * for use data and do whatever we desire with it.
  87. */
  88. perform_get_request(users_url)
  89.     .then((response) => {
  90.         names = response.reduce((plucked, user) => {
  91.             plucked.push(user.name);
  92.  
  93.             return plucked;
  94.         }, []);
  95.  
  96.         user_count = names.length;
  97.  
  98.         console.log(`${user_count} Names result: ${JSON.stringify(names)}`);
  99.     });
