Array Helper Methods: reduce

Next >>

In this lesson, I learnt about the reduce method which executes a reducer function (that you provide) on each element of the array, resulting in a single output value.

Code:

  1. let app = {};
  2.  
  3. app.numbers = [10, 20, 30, 40];
  4.  
  5. /*
  6. * The reduce helper executes a defined function on each element in the
  7. * array to give a single output. For example below we are summing up
  8. * all the integers in the array to give us a result of 100. The initial
  9. * value of the sum is set to 0 which is the second parameter of the reduce helper.
  10. *
  11. * More info: https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array/reduce
  12. */
  13. app.number_sum_result = app.numbers.reduce(function(sum, number) {
  14.     return sum + number;
  15. }, 0);
  16.  
  17. console.log('Number sum result: ' + app.number_sum_result);
  18.  
  19. app.primary_colors = [
  20.     {hex: '#FF0000', color: 'red'},
  21.     {hex: '#FFFF00', color: 'yellow'},
  22.     {hex: '#0000FF', color: 'blue'}
  23. ];
  24.  
  25. /*
  26. * In this example we use reduce to get all the values of colors and push them
  27. * to a new array of colors only.
  28. */
  29. app.colors_result = app.primary_colors.reduce(function(colors, primary) {
  30.     colors.push(primary.color);
  31.  
  32.     return colors;
  33. }, []);
  34.  
  35. console.log('Colors result: ' + JSON.stringify(app.colors_result));
  36.  
  37. app.parentheses = '(()()()())';
  38.  
  39. /*
  40. * In this example we check if a string of parentheses(above) are balanced. By this we mean
  41. * if all parentheses have an opening and closing bracket. All parentheses must have a starting
  42. * '(' symbol and end with ')' symbol.
  43. * 
  44. * Example solved in python: https://runestone.academy/runestone/books/published/pythonds/BasicDS/SimpleBalancedParentheses.html
  45. */
  46. app.balancedParens = function (string) {
  47.     let check = string.split('').reduce(function(counter, char) {
  48.         if (counter < 0) {
  49.             return counter;
  50.         } else if (char === '(') {
  51.             return ++counter;
  52.         } else if(char === ')') {
  53.             return --counter;
  54.         } else {
  55.             return counter;
  56.         }
  57.     }, 0);
  58.  
  59.     return check === 0 ? 'Balanced' : 'Not Balanced';
  60. };
  61.  
  62. console.log('Balanced parentheses check of "' + app.parentheses + '" result: ' + app.balancedParens(app.parentheses));
  63.  
  64. app.desks = [
  65.     { type: 'sitting' },
  66.     { type: 'standing' },
  67.     { type: 'sitting' },
  68.     { type: 'sitting' },
  69.     { type: 'standing' }
  70. ];
  71.  
  72. app.desk_types = app.desks.reduce(function(stats, desk) {
  73.     if(desk.type === 'sitting') {
  74.         stats.sitting += 1
  75.     } else {
  76.         stats.standing += 1
  77.     }
  78.  
  79.     return stats;
  80. }, { sitting: 0, standing: 0 });
  81.  
  82. console.log('Desk type stats: ' + JSON.stringify(app.desk_types));
  83.  
  84. app.duplicate_numbers = [1, 1, 2, 3, 4, 4];
  85.  
  86. /*
  87. * In this example we use reduce to remove duplicate numbers and only return unique numbers
  88. */
  89. app.uniqueNumbers = function (array) {
  90.     return array.reduce(function(unique_numbers, number) {
  91.         /*
  92.         * includes method is used here to check whether the number already exists in the result array
  93.         */
  94.         if (!unique_numbers.includes(number)) {
  95.             unique_numbers.push(number);
  96.         }
  97.  
  98.         return unique_numbers;
  99.     }, []);
  100. }
  101.  
  102. console.log('Unique numbers result: ' + JSON.stringify(app.uniqueNumbers(app.duplicate_numbers)));
