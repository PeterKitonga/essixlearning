Array Helper Methods: forEach

Next >>

In this lesson, I learnt about the forEach helper that is available to the Array object class. With this helper we can loop through an array without using for loops.

Code:

  1. let app = {};
  2.  
  3. app.colors = ['red', 'blue', 'green'];
  4.  
  5. app.colors.forEach(function(color, index) {
  6.     console.log('Color ' + (index + 1) + ': ' + color);
  7. });
  8.  
  9. // create an array of numbers
  10. app.numbers = [10, 52, 23, 44, 56];
  11.  
  12. // create a variable to hold the sum
  13. app.numbers_sum = 0;
  14.  
  15. // loop over the array, incrementing the sum variable
  16. app.numbers.forEach(function(number) {
  17.     let current_sum = app.numbers_sum;
  18.     app.numbers_sum += number;
  19.     console.log(number + ' added to the sum(' + current_sum + ') is: ' + app.numbers_sum);
  20. });
  21.  
  22. // print the sum variable
  23. console.log('Sum result is: ' + app.numbers_sum);
  24.  
  25. // can also work with objects
  26. app.images = [
  27.     { height: 10, width: 30 },
  28.     { height: 20, width: 90 },
  29.     { height: 54, width: 32 }
  30. ];
  31.  
  32. app.areas = [];
  33.  
  34. app.images.forEach(function(image, index) {
  35.     let area = image.height * image.width;
  36.     app.areas[index] = area; // push the area to the areas array
  37.     console.log('Area of image ' + (index + 1) + '(' + JSON.stringify(image) + ') is: ' + area);
  38. });
