Array Helper Methods: find

Next >>

In this lesson, I learnt about the find helper which like the filter function returns a new array that satisfies the condition in the callback function. The main difference is that it returns a single object which is the first result that satisfies the condition.

Code:

  1. let app = {};
  2.  
  3. app.users = [
  4.     {id: 11, name: 'Jill'},
  5.     {id: 23, name: 'Alex'},
  6.     {id: 48, name: 'Fred'},
  7.     {id: 55, name: 'Peter'},
  8.     {id: 56, name: 'Alex'}
  9. ];
  10.  
  11. /*
  12. * The find helper is similar to the filter helper, however, it retrieves
  13. * the first result that satisfies the condition in the callback. It returns
  14. * only one result.
  15. */
  16. app.user = app.users.find(function (user) {
  17.     return user.name === 'Alex';
  18. });
  19.  
  20. console.log('Result for the user: ' + JSON.stringify(app.user));
  21.  
  22. function Car(make, model) {
  23.     this.make = make;
  24.     this.model = model;
  25. };
  26.  
  27. /*
  28. * The helper can also be used on objects created from an object class.
  29. */
  30. app.cars = [
  31.     new Car('Ford', 'Focus'),
  32.     new Car('Toyota', 'Prius'),
  33.     new Car('Chevrolet', 'Camaro'),
  34.     new Car('Chevrolet', 'Corvette')
  35. ];
  36.  
  37. app.car = app.cars.find(function(car) {
  38.     return car.model === 'Camaro';
  39. });
  40.  
  41. console.log('Result for car model: ' + JSON.stringify(app.car));
  42.  
  43. app.posts = [
  44.     {id: 4, title: 'New Post'},
  45.     {id: 1, title: 'Old Post'},
  46. ];
  47.  
  48. app.comment = {post_id: 1, content: 'it was ok'};
  49.  
  50. app.post_for_comment = function (posts, comment) {
  51.     return posts.find(function(post) {
  52.         return post.id === comment.post_id;
  53.     });
  54. };
  55.  
  56. console.log('Result for post with the comment: ' + JSON.stringify(app.post_for_comment(app.posts, app.comment)));
  57.  
  58. app.ladders = [
  59.     {id: 1, height: 20},
  60.     {id: 3, height: 25}
  61. ];
  62.  
  63. /*
  64. * We can also use find to create a custom helper method for searching
  65. * through a result set depending on the property and value criteria defined
  66. */
  67. app.find_where = function (array, criteria) {
  68.     let property = Object.keys(criteria)[0];
  69.     let value = criteria[property];
  70.     let result = array.find(function(item) {
  71.         return item[property] === value;
  72.     });
  73.  
  74.     return result;
  75. }
  76.  
  77. console.log('Result of custom dynamic find helper: ' + JSON.stringify(app.find_where(app.ladders, {height: 25})));
