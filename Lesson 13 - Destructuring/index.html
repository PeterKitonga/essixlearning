Destructuring

Next >>

In this lesson, I learnt about destructuring which is used to deconstruct arrays(even objects) into individual variables during assignment of values.

Code:

  1. let app = {};
  2.  
  3. app.expense = {
  4.     type: 'Business',
  5.     amount: 'USD 45'
  6. };
  7.  
  8. /*
  9. * Destructuring is used to deconstruct arrays(even objects) into individual
  10. * variables during assignment of values. This means that the variable names
  11. * will automatically pick values from matching object key names or array indices.
  12. * For example, below the variable names match the object key names in the expense object.
  13. * This can be translated as:
  14. * const type = app.expense.type;
  15. * const amount = app.expense.amount;
  16. */
  17. const { type, amount } = app.expense;
  18.  
  19. console.log(`Value of variable 'type' is: ${type} and 'amount' is: ${amount}`);
  20.  
  21. app.list = [1, 2, 3, 4];
  22.  
  23. /*
  24. * Here, we match the variable indices to the value indices in the list array.
  25. * This can be translated as:
  26. * const a = app.list[0], b = app.list[2], c = app.list[3];
  27. * Note: Array index 1 has not be assigned to any variable.
  28. */
  29. const [ a, , b, c ] = app.list;
  30.  
  31. console.log(`Value for variable 'a' is: ${a}, 'b' is: ${b} and 'c' is ${c}`);
  32.  
  33. app.saved_file = {
  34.     name: 'repost',
  35.     extension: 'jpg',
  36.     size: 14040,
  37.     unit: 'MB'
  38. }
  39.  
  40. /*
  41. * Destructuring can also be used in function arguments where the parameter
  42. * passed is an object whose values will be used in the function. 
  43. */
  44. app.fileSummary = ({ name, extension, size, unit }, color) => {
  45.     return `The ${color} image file '${name}.${extension}' is of size ${size}${unit}`;
  46. };
  47.  
  48. console.log(app.fileSummary(app.saved_file, 'red'));
  49.  
  50. app.expenses = [
  51.     {
  52.         name: 'Government One',
  53.         tag: 'government',
  54.         amount: 'USD 15'
  55.     },
  56.     {
  57.         name: 'Individual Thirty',
  58.         tag: 'individual',
  59.         amount: 'USD 25'
  60.     },
  61.     {
  62.         name: 'Business Ten',
  63.         tag: 'business',
  64.         amount: 'USD 35'
  65.     },
  66.     {
  67.         name: 'Business Six',
  68.         tag: 'business',
  69.         amount: 'USD 45'
  70.     }
  71. ];
  72.  
  73. /*
  74. * We can also use the spread operator with destructuring. In the case
  75. * below, the spread operator will be assigned to the array objects 
  76. * after the first element.
  77. */
  78. const [ first, ...rest ] = app.expenses;
  79.  
  80. console.log(`The first expense is: ${JSON.stringify(first)} while the rest of the expenses are: ${JSON.stringify(rest)}`);
  81.  
  82. /*
  83. * Here, we use destructuring to get the tag of the first element in the
  84. * expenses array object.
  85. */
  86. const [ { tag } ] = app.expenses;
  87.  
  88. console.log(`The tag of the first expense is: ${tag}`);
  89.  
  90. app.google_offices = {
  91.     locations: ['Mountain View', 'New York', 'London']
  92. };
  93.  
  94. /*
  95. * Here, we fetch the first location in the locations array
  96. */
  97. const { locations: [ first_location ] } = app.google_offices;
  98.  
  99. console.log(`The first google office location is: ${first_location}`);
  100.  
  101. app.points = [
  102.     [4, 5],
  103.     [10, 1],
  104.     [12, 11],
  105.     [24, 7],
  106. ];
  107.  
  108. /*
  109. * To create an array of objects with x and y offsets we can use
  110. * destructuring and the map helper to generate the new array.
  111. */
  112. let graph = app.points.map(( [ x, y ] ) => {
  113.     return { x, y };
  114. });
  115.  
  116. console.log(`Graph data coordinates are: ${JSON.stringify(graph)}`);
  117.  
  118. const numbers = [1, 2, 3];
  119.  
  120. /*
  121. * In this example we want to double every element in the array via
  122. * a recursive function
  123. */
  124. function double(numbers) {
  125.     // we first fetch the first element in the array and the rest of the elements as well
  126.     const [ first_number, ...rest ] = numbers;
  127.     // get the length of the array passed
  128.     const { length } = numbers;
  129.  
  130.     // check if the recursion has gone beyond the bound of the array
  131.     if (length > 0) {
  132.         /*
  133.         * if still within the bounds of the array, double the first element
  134.         * and repeat the process
  135.         */
  136.         return [ first_number * 2, ...double(rest) ];
  137.     } else {
  138.         // otherwise if out of bounds return the default numbers
  139.         return numbers;
  140.     }
  141. }
  142.  
  143. let result = double(numbers);
  144.  
  145. console.log(`The result of double each element in array ${JSON.stringify(numbers)} is ${JSON.stringify(result)}`);
