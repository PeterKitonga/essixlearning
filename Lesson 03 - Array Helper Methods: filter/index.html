Array Helper Methods: filter

Next >>

In this lesson, I learnt about the filter helper which generates an new array that satisfies the condition set in the callback function. This helper is mostly used to filter and only return specific data related to another set of data.

Code:

  1. let app = {};
  2.  
  3. app.products = [
  4.     {name: 'cucumber', type: 'vegetable', quantity: 0, price: 1},
  5.     {name: 'banana', type: 'fruit', quantity: 10, price: 15},
  6.     {name: 'celery', type: 'vegetable', quantity: 30, price: 13},
  7.     {name: 'orange', type: 'fruit', quantity: 3, price: 5},
  8.     {name: 'green pepper', type: 'vegetable', quantity: 20, price: 2},
  9.     {name: 'garlic', type: 'vegetable', quantity: 2, price: 3},
  10.     {name: 'apple', type: 'fruit', quantity: 3, price: 25}
  11. ];
  12.  
  13. /*
  14. * This method creates a new array that passes the test within it's callback function.
  15. * It can be used to filter data that satisfy specific conditions. For example in the logic below
  16. * we only want products that have a type of 'fruit'
  17. */
  18. app.fruits = app.products.filter(function(product) {
  19.     return product.type === 'fruit';
  20. });
  21.  
  22. console.log('Result of fruits is: ' + JSON.stringify(app.fruits));
  23.  
  24. /*
  25. * Below is a more complex example where we want type is 'vegetable', 
  26. * quantity is greater than 0, price is less than 10
  27. */
  28. app.vegetables = app.products.filter(function(product) {
  29.     return product.type === 'vegetable' && product.quantity > 0 && product.price < 10;
  30. });
  31.  
  32. console.log('Result of vegetables is: ' + JSON.stringify(app.vegetables));
  33.  
  34. app.post = {id: 4, title: 'New Post'};
  35.  
  36. app.comments = [
  37.     {post_id: 4, content: 'awesome post'},
  38.     {post_id: 3, content: 'it was ok'},
  39.     {post_id: 2, content: 'very good'},
  40.     {post_id: 4, content: 'neat'}
  41. ];
  42.  
  43. /*
  44. * In this example we are filtering objects related to some other object. This 
  45. * can be useful for filtering related data from a database.
  46. */
  47. app.comments_for_post = function (post, comments) {
  48.     return comments.filter(function (comment) {
  49.         return comment.post_id === post.id;
  50.     });
  51. };
  52.  
  53. console.log('Result of comments for post (' + app.post.id + ') is: ' + JSON.stringify(app.comments_for_post(app.post, app.comments)));
