In this lesson, I learnt about generators and it's use cases. Generators are functions that allow us to enter and exit with values on demand at different points of execution. We would often use generators and for...of loops together since the for...of iterator takes advantage of the various features provided by generators.

Code:

  1. let app = {};
  2.  
  3. const colors = ['red', 'green', 'blue'];
  4.  
  5. /*
  6. * The for...of statements creates a loop iterating iterable objects such
  7. * as strings, arrays, typed arrays, maps, sets e.t.c.
  8. */
  9. for(const [index, color] of colors.entries()) {
  10.     console.log(`The color '${color}' is at index: '${index}'`);
  11. }
  12.  
  13. /*
  14. * Generators are functions that can return('yield') multiple values at different
  15. * execution times on demand. This is unlike regular functions that return a single value.
  16. * To define a generator we add an asterix after the function keyword. This means
  17. * they work great with iterators like the above for...of statement.
  18. */
  19. function* numbers() {
  20.     /*
  21.     * The yield statement in this point pauses execution
  22.     * of the function and returns the value. When next() is called again
  23.     * the function picks up from this yield to the next yield statement
  24.     */
  25.     yield 1;
  26.     // other code
  27.     yield 2;
  28.     // some other code
  29.     return 3;
  30. }
  31.  
  32. const number_generator = numbers();
  33.  
  34. /*
  35. * We use the next() method of the generator function to run it and fetch
  36. * the nearest returned('yield') value. The result of next is always an 
  37. * object with two properties:
  38. * 1. value - the yielded value.
  39. * 2. done - 'true' if the function code has finished, otherwise 'false'.
  40. */
  41. console.log(`Number generator first result: ${JSON.stringify(number_generator.next())}`);
  42. console.log(`Number generator second result: ${JSON.stringify(number_generator.next())}`);
  43. console.log(`Number generator third  result: ${JSON.stringify(number_generator.next())}`);
  44.  
  45. /*
  46. * Below we have an example where we simulate a person going to the
  47. * store to buy groceries and do laundry. The generator is not ran until we call next().
  48. * The function describes the whole journey where the next() method determines
  49. * when a desired action was performed
  50. */
  51. function* shopping() {
  52.     // moves to the sidewalk
  53.  
  54.     // walking down the sidewalk
  55.  
  56.     // go into the store with cash
  57.     const stuff_from_store = yield 'purchase';
  58.     /* here we pause and pickup on the next call where 
  59.     * yield is reassigned to the value 'groceries'
  60.     */
  61.  
  62.     // here we reassign yield to 'clean laundry' in the third call
  63.     const clean_clothes = yield 'bought groceries';
  64.  
  65.     // walking back home
  66.     return `Walks back home with ${stuff_from_store} and ${clean_clothes}`;
  67. }
  68.  
  69. // the journey
  70. const shopping_generator = shopping();
  71. // we start here 
  72. console.log(`${JSON.stringify(shopping_generator.next('pay with cash'))}`); // leaving our house
  73. // walked into the store
  74. // walking up and down the aisles..
  75. // purchase our stuff
  76. console.log(`${JSON.stringify(shopping_generator.next('groceries'))}`); // leaving the store with groceries
  77. // heads to the dry cleaner, cleans laundry and heads home
  78. console.log(`${JSON.stringify(shopping_generator.next('clean laundry'))}`); // heading home with groceries and clean laundry
  79.  
  80. function* _colors() {
  81.     yield 'red';
  82.     yield 'blue';
  83.     yield 'green';
  84. }
  85.  
  86. const primary_colors = [];
  87. /*
  88. * Here, we see a practical example of how the for...of loop
  89. * works with the generator above to fetch the values of yield
  90. * without using the next() method
  91. */
  92. for(const _color of _colors()) {
  93.     primary_colors.push(_color);
  94. }
  95.  
  96. console.log(`Colors generator result: ${JSON.stringify(primary_colors)}`);
  97.  
  98. const testing_team = {
  99.     lead: 'Amanda',
  100.     tester: 'Bill',
  101.     /**
  102.      * We use the Symbol.iterator here for key interpolation(dynamically created key)
  103.      * to tell the for...of loop how to iterate through this object. The asterix lets
  104.      * the for...of iterator know that this is a generator and should use the defined
  105.      * yields to get the values we need
  106.     */
  107.     *[Symbol.iterator]() {
  108.         yield this.lead;
  109.         yield this.tester;
  110.     }
  111. };
  112.  
  113. const engineering_team = {
  114.     testing_team, // we can add a reference to the testing team object
  115.     size: 4,
  116.     department: 'Engineering',
  117.     lead: 'Jill',
  118.     manager: 'Alex',
  119.     designer: 'Elsa',
  120.     engineer: 'Peter',
  121.     /**
  122.      * Here we use a generator to fetch the desired properties of
  123.      * this object including the testing team generator.
  124.      */
  125.     *[Symbol.iterator]() {
  126.         yield this.lead;
  127.         yield this.manager;
  128.         yield this.designer;
  129.         yield this.engineer;
  130.         /**
  131.          * Here, we delegate the testing team Symbol.iterator to this yield statement to allow
  132.          * the for...of loop to iterate through it as well. The asterix before the yield keyword
  133.          * signifies that there are extra yields that are to be taken into consideration.
  134.         */
  135.         yield* this.testing_team;
  136.     }
  137. };
  138.  
  139. const names = [];
  140.  
  141. /**
  142.  * We then use a for...of loop statement to iterate through the generator
  143.  * and push the yielded values to an array
  144. */
  145. for(const name of engineering_team) {
  146.     names.push(name);
  147. }
  148.  
  149. console.log(`Team member names are: ${JSON.stringify(names)}`);

Extra:

  1. /**
  2.  * Let's create a class that will create an comment object that
  3.  * has children comment objects. This will create a tree where each node
  4.  * has its child nodes
  5. */
  6. class Comment {
  7.     constructor(content, children) {
  8.         this.content = content;
  9.         this.children = children;
  10.     }
  11.  
  12.     /**
  13.      * Here we use this generator to flatten the tree structure
  14.      * and fetch the values we need(in this case the content)
  15.     */
  16.     *[Symbol.iterator]() {
  17.         yield this.content;
  18.  
  19.         /**
  20.          * Here we use a for...of loop since we cannot use array
  21.          * helpers in generators to yield values
  22.         */
  23.         for(let child of this.children) {
  24.             yield* child;
  25.         }
  26.     }
  27. }
  28.  
  29. const children = [
  30.     new Comment('good comment', []),
  31.     new Comment('bad comment', []),
  32.     new Comment('meh', []),
  33. ];
  34.  
  35. /**
  36.  * Here we create a tree with 'Great post' as the parent node
  37. */
  38. const tree = new Comment('Great post!', children);
  39.  
  40. console.log(`Tree result: ${JSON.stringify(tree)}`);
  41.  
  42. /**
  43.  * Here we will use the for...of loop to iterate through the generator 
  44.  * and get a flattened array
  45. */
  46. const values = [];
  47. for (const value of tree) {
  48.     values.push(value);
  49. }
  50.  
  51. console.log(`Tree values result: ${JSON.stringify(values)}`);
