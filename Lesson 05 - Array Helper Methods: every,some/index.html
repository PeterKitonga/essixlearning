Array Helper Methods: every,some

Next >>

In this lesson, I learnt about the every and some helpers that return a boolean depending on the test implemented in their callbacks. The 'every' method simply checks whether all elements of an array pass the test while the 'some' method check whether at least one item passes the test.

Code:

  1. let app = {};
  2.  
  3. app.computers = [
  4.     { name: 'Apple', ram: 24 },
  5.     { name: 'Compaq', ram: 4 },
  6.     { name: 'Acer', ram: 32 }
  7. ]; 
  8.  
  9. /*
  10. * The every helper tests whether all elements in a given array 
  11. * pass the test implememted in the callback function. This 
  12. * method returns a boolean.
  13. * 
  14. * More info: https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array/every
  15. */
  16. app.all_can_run_program = app.computers.every(function(computer) {
  17.     return computer.ram > 16;
  18. });
  19.  
  20. console.log('Every able computer check result: ' + app.all_can_run_program);
  21.  
  22. /*
  23. * The some helper tests whether at least one item in a give array
  24. * passes the test in the callback function. This method also returns
  25. * a boolean. This method will still return true if all items 
  26. * in the array satisfy the test.
  27. * 
  28. * More info: https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Array/some
  29. */
  30. app.some_can_run_program = app.computers.some(function(computer) {
  31.     return computer.ram > 16;
  32. });
  33.  
  34. console.log('Some able computers check result: ' + app.some_can_run_program);
  35.  
  36. app.names = [
  37.     "Alexandria",
  38.     "Matthew",
  39.     "Luke",
  40.     "Joe"
  41. ];
  42.  
  43. app.all_long_names = app.names.every(function(name) {
  44.     return name.length > 4;
  45. });
  46.  
  47. console.log('Every long name check result: ' + app.all_long_names);
  48.  
  49. app.some_long_names = app.names.some(function(name) {
  50.     return name.length > 4;
  51. });
  52.  
  53. console.log('Some long names check result: ' + app.some_long_names);
  54.  
  55. /*
  56. * In the below example we have a validate method that
  57. * checks whether all submitted fields of a form are filled.
  58. * Here, the every method is used to loop through and run the validator.
  59. */
  60. app.Field = function (value) {
  61.     this.value = value;
  62. }
  63.  
  64. app.fields = ['username', 'email', 'password'];
  65.  
  66. app.validate = function(field) {
  67.     return field.value.length > 0;
  68. }
  69.  
  70. app.username = new app.Field('kitongangugi');
  71. app.email = new app.Field('');
  72. app.password = new app.Field('password');
  73.  
  74. app.validate_check = app.fields.every(function(field) {
  75.     return app.validate(app[field]);
  76. });
  77.  
  78. console.log('Result for validation check: ' + app.validate_check);
  79.  
  80.
